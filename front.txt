server {
		listen 443 ssl;
		server_name teamn.sku-sku.com; #서브 도메인

		ssl_certificate /etc/letsencrypt/live/teamn.sku-sku.com/fullchain.pem;
		ssl_certificate_key /etc/letsencrypt/live/teamn.sku-sku.com/privkey.pem;

		location / {
					root /home/ubuntu/build; #실행할 프로젝트 경로
					index index.html index.htm; #리다이렉션 할 파일 명
					try_files $uri $uri/ =404; #/index.html; #요청된 url이 없을 경우 index.html을 반환
	}
}
